Sub PreviousMonth()
If Sheet18.Range("A2").Value = 1 Then
    Exit Sub
Else:
    Sheet18.Range("A2").Value = Sheet18.Range("A2").Value - 1
    Sheet18.Columns("P:BI").Hidden = True
    Sheet18.Range(Columns((Range("A2").Value - 1) + 15), Columns((Range("A2").Value - 1) + 26)).Hidden = False
    
   End If
End Sub
